    $str=substr(md5($wdcp_user.$wdcp_uid),8,6);
    //echo $str."<br>";
    if ($ul_str==0) {
        //echo $str;
        //echo $str."<br>";
        //$msg=$str."|".$wdcp_user."|".$wdcp_uid."|".$wdcp_gid."|".$wdcp_us;
        //file_put_contents(WD_ROOT."/data/1.txt",$msg);
        return md5($str.$wdcp_user.$wdcp_uid.$wdcp_gid.$wdcp_us);
    }else {
        //echo ;
